    string str;

    str = to_string(n);
    if (str.find('3') == string::npos) {
        return (false);
    }
    return (true);
}

int solution(int n) {
    int answer = 0;

    for (int idx = 1; idx <= n; idx++) {
        answer += 1;

        while (answer % 3 == 0 || isCharIncluded(answer)) {
            answer += 1;
        }
    }

    return answer;
}
