function solution(left, right) {
  let sum = 0;

  for (let num = left; num <= right; num++) {
    const sqrt = Math.sqrt(num);
    let cnt = 0;

    for (let i = 1; i <= sqrt; i++) {
      if (i === sqrt) {
        cnt++;
        continue;
      }
      if (num % i === 0) {
        cnt += 2;
      }
    }

    if (cnt % 2 === 0) {
      sum += num;
      continue;
    }

    sum -= num;
  }

  return sum;
}

function solution(left, right) {

      answer -= i;
    } else {
      answer += i;
    }
  }
  return answer;
