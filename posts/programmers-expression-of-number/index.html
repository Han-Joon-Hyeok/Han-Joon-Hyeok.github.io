function solution(n) {
  let point = 1;
  let sum = 0;
  let cnt = 0;
  for (let i = 1; point <= n; ) {
    sum += i;
    if (sum === n) {
      cnt++;
      sum = 0;
      i = ++point;
      continue;
    }
    if (sum > n) {
      i = ++point;
      sum = 0;
      continue;
    }
    i++;
  }
  return cnt;
}

function solution(n) {
  var answer = 0;
  function re(current, sum) {
    if (sum === n) return true;
    if (sum > n) return false;
    return re(current + 1, sum + current);
  }
  for (let i = 1; i <= n; i++) {
    if (re(i, 0)) answer++;
  }
  return answer;
}
