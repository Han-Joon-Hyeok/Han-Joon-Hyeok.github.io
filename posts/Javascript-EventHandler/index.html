let text = document.querySelector("body h2");
const Handler = {
    mouserOver: function(){
        text.innerHTML = "Mouse is on"
        text.style.color = "#ff9999"
    }
}

function init(){
    text.addEventListner("mouseOver", Handler.mouseOver);
}
