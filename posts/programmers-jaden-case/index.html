function solution(s) {
  const words = s.toLowerCase().split(" ");
  const jadenCase = words.map((word) => {
    const chars = word.split("");
    const head = chars[0];
    if (typeof head === "string") {
      chars.unshift(head.toUpperCase());
    }
    return chars.join("");
  });
  return jadenCase.join(" ");
}
